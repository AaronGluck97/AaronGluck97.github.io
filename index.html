My Own Int Parser (C#)

Always looking for better understanding, and making methods for practice always helps.

static int MyOwnParse(string word)
{
    if (String.IsNullOrEmpty(word))
        throw new Exception("not an int");

    int theNum = 0;
    bool negative = word[0]=='-';
    if (negative && word.Length == 1)
        throw new Exception("not an int");

    int end = negative ? 1 : 0;
    for (int i = word.Length - 1, j = 1; i >= end; i--, j *= 10)
    {
        if (word[i] < '0' || word[i] > '9')
            throw new Exception("not an int");

        theNum += (word[i] - '0') * j;
    }
    return negative ? -theNum : theNum;
}
